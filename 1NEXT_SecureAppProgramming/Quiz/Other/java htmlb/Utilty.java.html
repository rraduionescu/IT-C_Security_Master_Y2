package ro.ase.ism.java; import java.security.Provider; import java.security.Security; public class Utilty { public static boolean checkProvider(String providerName){ Provider provider = Security.getProvider(providerName); if(provider != null) return true; else return false; } public static char[] getHexFormat(byte[] value){ char[] symbols = "0123456789ABCDEF".toCharArray(); char[] result = new char[value.length * 2]; for(int i = 0;i<value.length;i++){ int temp = value[i] & 0xff; result[2*i+1] = symbols[temp & 0x0f]; result[2*i] = symbols[temp >> 4]; } return result; } }
